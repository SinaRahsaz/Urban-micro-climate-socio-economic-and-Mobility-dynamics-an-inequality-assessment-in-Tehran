import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Load the meteorological data
data = pd.read_csv("Meteorological Paramaters monthly.csv")

# Quick look at the data structure
print("First few rows of the dataset:")
print(data.head())
print("\nColumn names:")
print(data.columns)

# Function to calculate Gini coefficient
def calculate_gini(values):
    values = values.flatten()  # Ensure 1D array
    # Shift negative values to positive if any
    if np.min(values) < 0:
        values -= np.min(values)
    # Add small constant to avoid division by zero
    values += 1e-10
    sorted_values = np.sort(values)
    n = len(sorted_values)
    indices = np.arange(1, n + 1)
    # Gini formula: sum((2i - n - 1) * xi) / (n * sum(xi))
    gini = (np.sum((2 * indices - n - 1) * sorted_values)) / (n * np.sum(sorted_values))
    return gini

# Define meteorological parameters to analyze
parameters = [
    'temperature_2m_max (°C)', 
    'temperature_2m_min (°C)', 
    'temperature_2m_mean (°C)',
    'precipitation_sum (mm)', 
    'snowfall_sum (cm)', 
    'wind_speed_10m_max (km/h)'
]

# Create a DataFrame to store Gini coefficients by region
gini_data = pd.DataFrame({'Region': sorted(data['Region'].unique())})

# Calculate Gini coefficients for each parameter by region
for param in parameters:
    gini_values = []
    for region in gini_data['Region']:
        # Extract data for the given region and parameter
        region_values = data[data['Region'] == region][param].values
        gini_value = calculate_gini(region_values)
        gini_values.append(gini_value)
    gini_data[param] = gini_values

# Set up the plot
plt.figure(figsize=(10, 6))

# Define colors and markers for visual distinction
colors = ['deepskyblue', 'darkorange', 'limegreen', 'mediumorchid', 'crimson', 'saddlebrown']
markers = ['o', 's', '^', 'v', 'D', 'P']
labels = [
    'Max Temperature (°C)', 
    'Min Temperature (°C)', 
    'Mean Temperature (°C)',
    'Precipitation (mm)', 
    'Snowfall (cm)', 
    'Max Wind Speed (km/h)'
]

# Plot Gini coefficients for each parameter
for idx, param in enumerate(parameters):
    plt.plot(gini_data['Region'], gini_data[param], 
             label=labels[idx], color=colors[idx], marker=markers[idx])

# Customize the plot
plt.title("Gini Coefficients for Meteorological Parameters by Region (2016–2022)")
plt.xlabel("Region")
plt.ylabel("Gini Coefficient")
plt.xticks(gini_data['Region'])
plt.legend(title="Parameters", bbox_to_anchor=(1.05, 1), loc='upper left')
plt.grid(True, linestyle='--', alpha=0.7)
plt.tight_layout()
plt.show()
