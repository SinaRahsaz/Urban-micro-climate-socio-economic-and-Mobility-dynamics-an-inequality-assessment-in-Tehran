import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Load the socio-economic mobility data
data = pd.read_csv("SEM data monthly.csv")

# List of socio-economic mobility indicators
indicators = [
    'Fixed Assets Rate',
    'Women Occupation Rate',
    'Men Occupation Rate',
    'Operating Sale Revenue Rate',
    'Participation Rate',
    'Trip Attraction Rate',
    'Trip Generation Rate',
    'Housing Price per meter'
]

# Function to calculate Gini coefficient
def calculate_gini(values):
    values = np.array(values, dtype=float)
    # Shift negative values to positive if any
    if np.min(values) < 0:
        values -= np.min(values)
    # Add small constant to avoid division by zero
    values += 1e-10
    sorted_values = np.sort(values)
    n = len(sorted_values)
    indices = np.arange(1, n + 1)
    # Gini formula: sum((2i - n - 1) * xi) / (n * sum(xi))
    gini = (np.sum((2 * indices - n - 1) * sorted_values)) / (n * np.sum(sorted_values))
    return gini

# Create a DataFrame with unique regions
region_gini = pd.DataFrame({'Region': sorted(data['Region'].unique())})

# Calculate Gini coefficients for each indicator by region
for indicator in indicators:
    gini_values = []
    for region in region_gini['Region']:
        # Extract data for the given region and indicator
        region_values = data[data['Region'] == region][indicator].astype(float).values
        gini_value = calculate_gini(region_values)
        gini_values.append(gini_value)
    region_gini[indicator] = gini_values

# Plot the Gini coefficients by region
plt.figure(figsize=(16, 10))
for indicator in indicators:
    plt.plot(region_gini['Region'], region_gini[indicator], marker='o', label=indicator)

# Customize the plot
plt.title("Gini Coefficients for Socio-Economic Mobility Indicators by Region (20162022)", fontsize=16)
plt.xlabel("Region", fontsize=14)
plt.ylabel("Gini Coefficient", fontsize=14)
plt.xticks(region_gini['Region'])
plt.ylim(0, 0.6)
plt.grid(True, linestyle='--', alpha=0.5)
plt.legend(title="Indicators", bbox_to_anchor=(1.05, 1), loc='upper left')
plt.tight_layout()
plt.show()
