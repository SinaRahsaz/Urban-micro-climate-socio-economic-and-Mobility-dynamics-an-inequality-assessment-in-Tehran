import pandas as pd
import numpy as np
import matplotlib.pyplot as plt

# Load the socio-economic mobility data
data = pd.read_csv("SEM data monthly.csv")

# Convert Year and Region columns to integers for consistency
data['Year'] = data['Year'].astype(int)
data['Region'] = data['Region'].astype(int)

# List of socio-economic mobility indicators
indicators = [
    'Fixed Assets Rate',
    'Women Occupation Rate',
    'Men Occupation Rate',
    'Operating Sale Revenue Rate',
    'Participation Rate',
    'Trip Attraction Rate',
    'Trip Generation Rate',
    'Housing Price per meter'
]

# Function to calculate Gini coefficient
def calculate_gini(values):
    values = np.array(values, dtype=float)
    # Shift negative values to positive if any
    if np.min(values) < 0:
        values -= np.min(values)
    # Add small constant to avoid division by zero
    values += 1e-10
    sorted_values = np.sort(values)
    n = len(sorted_values)
    indices = np.arange(1, n + 1)
    # Gini formula: sum((2i - n - 1) * xi) / (n * sum(xi))
    gini = (np.sum((2 * indices - n - 1) * sorted_values)) / (n * np.sum(sorted_values))
    return gini

# Get unique years and sort them
years = sorted(data['Year'].unique())

# Create a DataFrame to store Gini coefficients for each year
gini_data = pd.DataFrame({'Year': years})

# Calculate Gini coefficients for each indicator across all regions per year
for indicator in indicators:
    yearly_gini = []
    for year in years:
        # Extract data for the given year and indicator
        year_values = data[data['Year'] == year][indicator].astype(float).values
        gini_value = calculate_gini(year_values)
        yearly_gini.append(gini_value)
    gini_data[indicator] = yearly_gini

# Plot the Gini coefficients over time
plt.figure(figsize=(16, 10))
for indicator in indicators:
    plt.plot(gini_data['Year'], gini_data[indicator], marker='o', label=indicator)

# Customize the plot
plt.title("Gini Coefficients for Socio-Economic Mobility Indicators (2016–2022)", fontsize=16)
plt.xlabel("Year", fontsize=14)
plt.ylabel("Gini Coefficient", fontsize=14)
plt.xticks(years)
plt.ylim(0, 0.6)
plt.grid(True, linestyle='--', alpha=0.5)
plt.legend(title="Indicators", bbox_to_anchor=(1.05, 1), loc='upper left')
plt.tight_layout()
plt.show()
